#include<bits/stdc++.h>

using namespace std;

class StackQueue{
private:
    stack<int> s1;
    stack<int> s2;
public:
    void push(int B);
    int pop();

};

void StackQueue :: push(int x){
    s1.push(x);
}

int StackQueue :: pop(){
    if(s1.size() == 0) return -1;
    while(s1.size() > 1){
        s2.push(s1.top());
        s1.pop();
    }
    int pop = s1.top(); s1.pop();
    while(s2.size()){
        s1.push(s2.top());
        s2.pop();
    }
    return pop;
}
